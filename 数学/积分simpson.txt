double simpson (double l , double r , double all , double eps) {  
    double m = (l + r) / 2.0;  
    double L = simpson (l , m) , R = simpson (m , r);  
    if (fabs (L + R - all) <= 15 * eps) return L + R + (L + R - all) / 15;  
    return simpson (l , m , L , eps / 2.0) + simpson (m , r , R , eps / 2.0);  
}
